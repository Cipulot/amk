cmake_minimum_required(VERSION 3.15)

project(astro VERSION 0.0.1
            DESCRIPTION "Keyboard firmware")

set(CMAKE_SYSTEM_NAME  Generic)
set(CMAKE_SYSTEM_PROCESSOR ARM)
set(CMAKE_CROSSCOMPILING 1)

set(CMAKE_C_COMPILER arm-none-eabi-gcc CACHE PATH "" FORCE)
set(CMAKE_C_FLAGS  "-std=gnu99 -DNRF52840_XXAA -DCONFIG_GPIO_AS_PINRESET" CACHE INTERNAL "C Compiler options")

add_subdirectory(tinyusb)
add_subdirectory(tmk)
add_subdirectory(main)
add_subdirectory(keyboards)
